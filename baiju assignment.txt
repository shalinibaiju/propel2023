/*create employee*/
create table employee
(
empno number(3) primary key,
managerid number(3),
firstname varchar(20) not null,
lastname varchar(20),
userid varchar(20),
deptno number(2),
salary number(5,2),
commisssion number(3),
joiningdate date,
designation varchar(25),
constraint fk_10 foreign key(deptno) references department(deptno)
ON DELETE CASCADE
);
desc employee
